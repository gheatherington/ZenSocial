    **1. Create `Scripts/Instagram/notification-bridge.js`:**

    A JS script injected at `atDocumentEnd` (after DOM is ready) that:
    - Observes Instagram's notification badge/indicator in the DOM (the notification bell icon typically has a badge dot or count)
    - Periodically checks (every 30 seconds while page is visible) for new notification indicators
    - When new notifications are detected, uses `window.webkit.messageHandlers.zenNotification.postMessage()` to send notification data to native
    - Extracts available info: notification count/badge presence, and if possible, notification text from the activity feed

    Implementation approach (Claude's discretion per CONTEXT.md):
    - Instagram's web notification indicator is typically on the heart/activity icon in the nav bar
    - Look for notification badge elements: elements with `aria-label` containing "notification" or activity-related text, or red dot badge indicators
    - For richer data: attempt to read from Instagram's activity page (`/accounts/activity/`) if the user navigates there, extracting notification items
    - Use a MutationObserver on the nav bar area to detect badge changes in real-time (complement to polling interval)
    - Send message format: `{ type: "badge_change", hasNew: true, count: N }` or `{ type: "activity_item", text: "user liked your photo", url: "/p/ABC123/" }`

    ```javascript
    (function() {
        'use strict';

        const POLL_INTERVAL_MS = 30000;
        let lastKnownBadge = false;
        let observer = null;

        function findNotificationBadge() {
            // Instagram web uses various indicators for notifications
            // Look for the activity/heart icon's badge indicator
            const navItems = document.querySelectorAll('[role="navigation"] a, nav a');
            for (const item of navItems) {
                const href = item.getAttribute('href') || '';
                if (href.includes('/accounts/activity') || href.includes('/direct/inbox')) {
                    // Check for badge dot (usually a child element with specific styling)
                    const badge = item.querySelector('[aria-label*="notification"], [aria-label*="new"], [class*="badge"], [class*="dot"]');
                    if (badge) return { hasNew: true, element: item };
                }
            }
            // Fallback: check for any red dot indicators in nav area
            const dots = document.querySelectorAll('nav [style*="background-color: rgb(255, 48, 64)"], nav [style*="background: rgb(255, 48, 64)"]');
            if (dots.length > 0) return { hasNew: true, element: null };
            return { hasNew: false, element: null };
        }

        function checkAndNotify() {
            if (!window.webkit?.messageHandlers?.zenNotification) return;
            const result = findNotificationBadge();
            if (result.hasNew && !lastKnownBadge) {
                window.webkit.messageHandlers.zenNotification.postMessage({
                    type: 'badge_change',
                    hasNew: true,
                    timestamp: Date.now()
                });
            }
            lastKnownBadge = result.hasNew;
        }

        // Poll periodically
        setInterval(checkAndNotify, POLL_INTERVAL_MS);

        // Also observe DOM mutations in nav area for real-time detection
        function setupObserver() {
            const nav = document.querySelector('nav, [role="navigation"]');
            if (!nav) {
                // Retry until nav exists
                setTimeout(setupObserver, 2000);
                return;
            }
            observer = new MutationObserver(() => {
                checkAndNotify();
            });
            observer.observe(nav, { childList: true, subtree: true, attributes: true });
        }

        // Initial check after page settles
        setTimeout(() => {
            checkAndNotify();
            setupObserver();
        }, 3000);
    })();
    ```

    NOTE: Instagram's DOM changes frequently. The selectors above are starting points. The executor should use Safari Web Inspector connected to the simulator to verify which selectors actually match Instagram's current notification badge. The script should be resilient -- if no badge element is found, it silently does nothing (no errors). Claude's discretion on exact selectors.

    **4. Create `ZenSocial/Services/NotificationPoller.swift`:**

    An `@MainActor` class that:
    - Conforms to `WKScriptMessageHandler` to receive messages from the JS bridge (foreground detection)
    - Implements background polling via URLSession using exported Instagram session cookies from `DataStoreManager`
    - Maintains last-known notification state in `UserDefaults` for diffing
    - Schedules local `UNNotificationRequest` via `UNUserNotificationCenter` only when new activity is detected
    - Registers a `BGAppRefreshTask` for background polling
    - Respects `NotificationManager.shared.userWantsNotifications` toggle

    ```swift
    import Foundation
    import WebKit
    import UserNotifications
    import BackgroundTasks
    import os

    @MainActor
    class NotificationPoller: NSObject, WKScriptMessageHandler {
        static let shared = NotificationPoller()
        static let bgTaskIdentifier = "com.zensocial.notification-check"

        private let logger = Logger(subsystem: "com.zensocial.app", category: "NotificationPoller")
        private var lastNotificationTimestamp: TimeInterval = 0

        /// Key for UserDefaults: stores the last-known Instagram notification badge hash/count
        private let lastKnownStateKey = "instagramLastKnownNotificationState"

        override init() {
            super.init()
        }

        // MARK: - WKScriptMessageHandler (Foreground Detection)

        // Receives messages from notification-bridge.js
        nonisolated func userContentController(
            _ userContentController: WKUserContentController,
            didReceive message: WKScriptMessage
        ) {
            guard message.name == "zenNotification",
                  let body = message.body as? [String: Any],
                  let type = body["type"] as? String
            else { return }

            Task { @MainActor in
                guard NotificationManager.shared.userWantsNotifications,
                      NotificationManager.shared.authorizationStatus == .authorized
                else { return }

                switch type {
                case "badge_change":
                    let timestamp = body["timestamp"] as? TimeInterval ?? Date().timeIntervalSince1970
                    guard timestamp > lastNotificationTimestamp else { return }
                    lastNotificationTimestamp = timestamp
                    await scheduleLocalNotification(
                        title: "Instagram",
                        body: "You have new activity on Instagram",
                        url: "https://www.instagram.com/accounts/activity/"
                    )
                default:
                    break
                }
            }
        }

        // MARK: - Local Notification Scheduling

        func scheduleLocalNotification(title: String, body: String, url: String) async {
            let content = UNMutableNotificationContent()
            content.title = title
            content.body = body
            content.sound = .default
            content.userInfo = ["instagram_url": url]
            content.threadIdentifier = "instagram_activity"

            let request = UNNotificationRequest(
                identifier: "instagram-\(Date().timeIntervalSince1970)",
                content: content,
                trigger: nil  // Deliver immediately
            )

            do {
                try await UNUserNotificationCenter.current().add(request)
                logger.info("Scheduled local notification: \(body)")
            } catch {
                logger.error("Failed to schedule notification: \(error.localizedDescription)")
            }
        }

        // MARK: - Instagram Cookie Export

        /// Exports Instagram session cookies from WKHTTPCookieStore for use with URLSession.
        /// This allows background polling without WKWebView (which is suspended in background).
        func exportInstagramCookies() async -> [HTTPCookie] {
            let dataStore = DataStoreManager.dataStore(for: .instagram)
            let cookies = await dataStore.httpCookieStore.allCookies()
            return cookies.filter { cookie in
                cookie.domain.contains("instagram.com")
            }
        }

        // MARK: - Background Instagram Notification Check

        /// Checks Instagram's notification state using URLSession with exported session cookies.
        /// Returns true if new activity was detected compared to last-known state.
        func checkInstagramNotifications() async -> Bool {
            let cookies = await exportInstagramCookies()
            guard !cookies.isEmpty else {
                logger.warning("No Instagram cookies available for background poll")
                return false
            }

            // Create a URLSession configuration with Instagram's cookies
            let sessionConfig = URLSessionConfiguration.ephemeral
            let cookieStorage = HTTPCookieStorage()
            for cookie in cookies {
                cookieStorage.setCookie(cookie)
            }
            sessionConfig.httpCookieStorage = cookieStorage
            sessionConfig.timeoutIntervalForRequest = 15

            let session = URLSession(configuration: sessionConfig)

            // Fetch Instagram's web notification endpoint.
            // Instagram's web app uses GraphQL for notification data. A lightweight approach
            // is to fetch the main page and check for notification indicators in the HTML,
            // or hit the activity endpoint and look for new items.
            //
            // Strategy: GET instagram.com and parse the HTML for notification badge indicators
            // (the initial HTML includes some serialized data about notification state).
            // Alternatively, use the GraphQL endpoint that the notification badge polls.
            //
            // We use the main page approach as it's more stable across Instagram updates.
            guard let url = URL(string: "https://www.instagram.com/") else { return false }

            var request = URLRequest(url: url)
            request.setValue("Mozilla/5.0 (iPhone; CPU iPhone OS 17_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/17.0 Mobile/15E148 Safari/604.1", forHTTPHeaderField: "User-Agent")
            request.setValue("text/html", forHTTPHeaderField: "Accept")

            do {
                let (data, response) = try await session.data(for: request)

                guard let httpResponse = response as? HTTPURLResponse,
                      httpResponse.statusCode == 200,
                      let html = String(data: data, encoding: .utf8)
                else {
                    logger.warning("Instagram background fetch failed or returned non-200")
                    return false
                }

                // Look for notification indicators in the HTML/serialized data.
                // Instagram embeds JSON data in <script> tags that includes notification badge state.
                // We look for indicators like "badge_count" or "has_unseen" in the serialized JSON.
                let hasNotifications = detectNotificationsInHTML(html)

                // Compare against last-known state
                let lastKnownState = UserDefaults.standard.bool(forKey: lastKnownStateKey)

                if hasNotifications && !lastKnownState {
                    // New notifications detected
                    UserDefaults.standard.set(true, forKey: lastKnownStateKey)
                    return true
                } else if !hasNotifications {
                    // No notifications -- update state
                    UserDefaults.standard.set(false, forKey: lastKnownStateKey)
                }

                return false
            } catch {
                logger.error("Instagram background fetch error: \(error.localizedDescription)")
                return false
            }
        }

        /// Parses Instagram's HTML response for notification badge indicators.
        /// Instagram embeds serialized JSON in the initial page load that includes notification state.
        private func detectNotificationsInHTML(_ html: String) -> Bool {
            // Strategy 1: Look for badge_count in serialized JSON
            if html.contains("\"badge_count\":") {
                // Extract the badge_count value
                if let range = html.range(of: "\"badge_count\":\\s*(\\d+)", options: .regularExpression) {
                    let match = String(html[range])
                    let digits = match.filter { $0.isNumber }
                    if let count = Int(digits), count > 0 {
                        return true
                    }
                }
            }

            // Strategy 2: Look for has_unseen notification indicators
            if html.contains("\"has_unseen\":true") || html.contains("\"unseen_count\":") {
                if let range = html.range(of: "\"unseen_count\":\\s*(\\d+)", options: .regularExpression) {
                    let match = String(html[range])
                    let digits = match.filter { $0.isNumber }
                    if let count = Int(digits), count > 0 {
                        return true
                    }
                }
                // has_unseen:true without a count still indicates activity
                if html.contains("\"has_unseen\":true") {
                    return true
                }
            }

            return false
        }

        // MARK: - BGAppRefreshTask

        func registerBackgroundTask() {
            BGTaskScheduler.shared.register(
                forTaskWithIdentifier: Self.bgTaskIdentifier,
                using: nil
            ) { task in
                guard let refreshTask = task as? BGAppRefreshTask else { return }
                Task { @MainActor in
                    await self.handleBackgroundRefresh(task: refreshTask)
                }
            }
        }

        func scheduleBackgroundRefresh() {
            let request = BGAppRefreshTaskRequest(identifier: Self.bgTaskIdentifier)
            request.earliestBeginDate = Date(timeIntervalSinceNow: 15 * 60) // 15 minutes
            do {
                try BGTaskScheduler.shared.submit(request)
                logger.info("Scheduled background refresh")
            } catch {
                logger.error("Failed to schedule background refresh: \(error.localizedDescription)")
            }
        }

        private func handleBackgroundRefresh(task: BGAppRefreshTask) async {
            // Schedule next refresh immediately
            scheduleBackgroundRefresh()

            guard NotificationManager.shared.userWantsNotifications,
                  NotificationManager.shared.authorizationStatus == .authorized
            else {
                task.setTaskCompleted(success: true)
                return
            }

            // Set expiration handler
            task.expirationHandler = {
                task.setTaskCompleted(success: false)
            }

            // Use URLSession with exported Instagram cookies to check for new notifications.
            // This is the on-device approach: export cookies from DataStoreManager's
            // WKHTTPCookieStore, make a lightweight HTTP request to Instagram, and
            // compare against last-known notification state.
            let hasNewActivity = await checkInstagramNotifications()

            if hasNewActivity {
                await scheduleLocalNotification(
                    title: "Instagram",
                    body: "You have new activity on Instagram",
                    url: "https://www.instagram.com/accounts/activity/"
                )
            }

            task.setTaskCompleted(success: true)
        }
    }
    ```

    Key design notes for background polling (addresses PUSH-02):
    - WKWebView cannot execute JS while the app is suspended. Background polling MUST use URLSession.
    - Instagram session cookies are exported from `DataStoreManager.dataStore(for: .instagram).httpCookieStore` via `allCookies()`.
    - These cookies are injected into a fresh `URLSession` configuration to make authenticated requests to Instagram.
    - The response HTML is parsed for notification badge indicators (Instagram embeds notification state in serialized JSON on the main page).
    - A last-known state is stored in `UserDefaults` to diff against. Only NEW notifications trigger a local notification.
    - This avoids the anti-pattern of unconditional "you may have new activity" notifications on every BGAppRefreshTask cycle.

    NOTE: Instagram's HTML structure and embedded JSON format may change. The `detectNotificationsInHTML` parser uses multiple strategies (badge_count, has_unseen, unseen_count) for resilience. The executor should verify these patterns against current Instagram HTML using Safari Web Inspector. If none match, fall back to a conservative approach that does not fire false-positive notifications.
