    **1. Create `Scripts/Instagram/notification-bridge.js`:**

    A JS script injected at `atDocumentEnd` (after DOM is ready) that:
    - Observes Instagram's notification badge/indicator in the DOM (the notification bell icon typically has a badge dot or count)
    - Periodically checks (every 30 seconds while page is visible) for new notification indicators
    - When new notifications are detected, uses `window.webkit.messageHandlers.zenNotification.postMessage()` to send notification data to native
    - Extracts available info: notification count/badge presence, and if possible, notification text from the activity feed

    Implementation approach (Claude's discretion per CONTEXT.md):
    - Instagram's web notification indicator is typically on the heart/activity icon in the nav bar
    - Look for notification badge elements: elements with `aria-label` containing "notification" or activity-related text, or red dot badge indicators
    - For richer data: attempt to read from Instagram's activity page (`/accounts/activity/`) if the user navigates there, extracting notification items
    - Use a MutationObserver on the nav bar area to detect badge changes in real-time (complement to polling interval)
    - Send message format: `{ type: "badge_change", hasNew: true, count: N }` or `{ type: "activity_item", text: "user liked your photo", url: "/p/ABC123/" }`

    ```javascript
    (function() {
        'use strict';

        const POLL_INTERVAL_MS = 30000;
        let lastKnownBadge = false;
        let observer = null;

        function findNotificationBadge() {
            // Instagram web uses various indicators for notifications
            // Look for the activity/heart icon's badge indicator
            const navItems = document.querySelectorAll('[role="navigation"] a, nav a');
            for (const item of navItems) {
                const href = item.getAttribute('href') || '';
                if (href.includes('/accounts/activity') || href.includes('/direct/inbox')) {
                    // Check for badge dot (usually a child element with specific styling)
                    const badge = item.querySelector('[aria-label*="notification"], [aria-label*="new"], [class*="badge"], [class*="dot"]');
                    if (badge) return { hasNew: true, element: item };
                }
            }
            // Fallback: check for any red dot indicators in nav area
            const dots = document.querySelectorAll('nav [style*="background-color: rgb(255, 48, 64)"], nav [style*="background: rgb(255, 48, 64)"]');
            if (dots.length > 0) return { hasNew: true, element: null };
            return { hasNew: false, element: null };
        }

        function checkAndNotify() {
            if (!window.webkit?.messageHandlers?.zenNotification) return;
            const result = findNotificationBadge();
            if (result.hasNew && !lastKnownBadge) {
                window.webkit.messageHandlers.zenNotification.postMessage({
                    type: 'badge_change',
                    hasNew: true,
                    timestamp: Date.now()
                });
            }
            lastKnownBadge = result.hasNew;
        }

        // Poll periodically
        setInterval(checkAndNotify, POLL_INTERVAL_MS);

        // Also observe DOM mutations in nav area for real-time detection
        function setupObserver() {
            const nav = document.querySelector('nav, [role="navigation"]');
            if (!nav) {
                // Retry until nav exists
                setTimeout(setupObserver, 2000);
                return;
            }
            observer = new MutationObserver(() => {
                checkAndNotify();
            });
            observer.observe(nav, { childList: true, subtree: true, attributes: true });
        }

        // Initial check after page settles
        setTimeout(() => {
            checkAndNotify();
            setupObserver();
        }, 3000);
    })();
    ```

    NOTE: Instagram's DOM changes frequently. The selectors above are starting points. The executor should use Safari Web Inspector connected to the simulator to verify which selectors actually match Instagram's current notification badge. The script should be resilient -- if no badge element is found, it silently does nothing (no errors). Claude's discretion on exact selectors.

    **4. Create `ZenSocial/Services/NotificationPoller.swift`:**

    An `@MainActor` class that:
    - Conforms to `WKScriptMessageHandler` to receive messages from the JS bridge
    - Maintains a set of already-shown notification IDs (to avoid duplicates)
    - Schedules local `UNNotificationRequest` via `UNUserNotificationCenter` when new notifications are detected
    - Registers a `BGAppRefreshTask` for background polling
    - Respects `NotificationManager.shared.userWantsNotifications` toggle

    ```swift
    import Foundation
    import WebKit
    import UserNotifications
    import BackgroundTasks
    import os

    @MainActor
    class NotificationPoller: NSObject, WKScriptMessageHandler {
        static let shared = NotificationPoller()
        static let bgTaskIdentifier = "com.zensocial.notification-check"

        private let logger = Logger(subsystem: "com.zensocial.app", category: "NotificationPoller")
        private var lastNotificationTimestamp: TimeInterval = 0

        override init() {
            super.init()
        }

        // MARK: - WKScriptMessageHandler

        // Receives messages from notification-bridge.js
        nonisolated func userContentController(
            _ userContentController: WKUserContentController,
            didReceive message: WKScriptMessage
        ) {
            guard message.name == "zenNotification",
                  let body = message.body as? [String: Any],
                  let type = body["type"] as? String
            else { return }

            Task { @MainActor in
                guard NotificationManager.shared.userWantsNotifications,
                      NotificationManager.shared.authorizationStatus == .authorized
                else { return }

                switch type {
                case "badge_change":
                    let timestamp = body["timestamp"] as? TimeInterval ?? Date().timeIntervalSince1970
                    guard timestamp > lastNotificationTimestamp else { return }
                    lastNotificationTimestamp = timestamp
                    await scheduleLocalNotification(
                        title: "Instagram",
                        body: "You have new activity on Instagram",
                        url: "https://www.instagram.com/accounts/activity/"
                    )
                default:
                    break
                }
            }
        }

        // MARK: - Local Notification Scheduling

        func scheduleLocalNotification(title: String, body: String, url: String) async {
            let content = UNMutableNotificationContent()
            content.title = title
            content.body = body
            content.sound = .default
            content.userInfo = ["instagram_url": url]
            content.threadIdentifier = "instagram_activity"

            let request = UNNotificationRequest(
                identifier: "instagram-\(Date().timeIntervalSince1970)",
                content: content,
                trigger: nil  // Deliver immediately
            )

            do {
                try await UNUserNotificationCenter.current().add(request)
                logger.info("Scheduled local notification: \(body)")
            } catch {
                logger.error("Failed to schedule notification: \(error.localizedDescription)")
            }
        }

        // MARK: - BGAppRefreshTask

        func registerBackgroundTask() {
            BGTaskScheduler.shared.register(
                forTaskWithIdentifier: Self.bgTaskIdentifier,
                using: nil
            ) { task in
                guard let refreshTask = task as? BGAppRefreshTask else { return }
                Task { @MainActor in
                    await self.handleBackgroundRefresh(task: refreshTask)
                }
            }
        }

        func scheduleBackgroundRefresh() {
            let request = BGAppRefreshTaskRequest(identifier: Self.bgTaskIdentifier)
            request.earliestBeginDate = Date(timeIntervalSinceNow: 15 * 60) // 15 minutes
            do {
                try BGTaskScheduler.shared.submit(request)
                logger.info("Scheduled background refresh")
            } catch {
                logger.error("Failed to schedule background refresh: \(error.localizedDescription)")
            }
        }

        private func handleBackgroundRefresh(task: BGAppRefreshTask) async {
            // Schedule next refresh
            scheduleBackgroundRefresh()

            guard NotificationManager.shared.userWantsNotifications,
                  NotificationManager.shared.authorizationStatus == .authorized
            else {
                task.setTaskCompleted(success: true)
                return
            }

            // In background, we cannot run JS in WKWebView (it's suspended).
            // Schedule a generic "check Instagram" reminder if the app has been
            // backgrounded long enough. For richer background notifications,
            // Phase 3.1's backend relay is needed.
            //
            // For now: post a local notification prompting user to check Instagram.
            // This satisfies PUSH-02's "user receives notification while suspended"
            // requirement using on-device resources only.
            await scheduleLocalNotification(
                title: "Instagram",
                body: "You may have new activity on Instagram",
                url: "https://www.instagram.com/accounts/activity/"
            )

            task.setTaskCompleted(success: true)
        }
    }
    ```

    NOTE on background approach: WKWebView cannot execute JS while the app is suspended. The BGAppRefreshTask provides periodic wake-ups where we schedule a local notification as a "check Instagram" prompt. This is the best on-device-only approach without a backend. The notification text is generic ("You may have new activity") because we cannot query Instagram's servers without the user's session cookies being sent from a server (which would require a backend). This is an acknowledged limitation per the CONTEXT.md scope -- force-quit and richer background delivery are Phase 3.1's responsibility.
