    **1. Create `ZenSocial/Services/NotificationManager.swift`:**

    An `@Observable @MainActor` class that wraps `UNUserNotificationCenter`:

    ```swift
    import Foundation
    import UserNotifications
    import Observation
    import UIKit

    @Observable
    @MainActor
    class NotificationManager: NSObject, UNUserNotificationCenterDelegate {
        // Published state
        var authorizationStatus: UNAuthorizationStatus = .notDetermined
        var hasCompletedPrePrompt: Bool = false

        // Persisted: has the pre-prompt been shown at least once?
        @ObservationIgnored
        @AppStorage("notificationPrePromptShown") private var prePromptShown: Bool = false

        // Persisted: user's in-app toggle preference
        @ObservationIgnored
        @AppStorage("notificationsEnabled") var userWantsNotifications: Bool = true

        static let shared = NotificationManager()

        override init() {
            super.init()
            hasCompletedPrePrompt = prePromptShown
        }

        // Call at app launch to read current authorization state
        func refreshAuthorizationStatus() async {
            let settings = await UNUserNotificationCenter.current().notificationSettings()
            authorizationStatus = settings.authorizationStatus
        }

        // Two-stage permission: pre-prompt accepted -> request system dialog
        func requestPermission() async -> Bool {
            let settings = await UNUserNotificationCenter.current().notificationSettings()
            authorizationStatus = settings.authorizationStatus

            guard settings.authorizationStatus == .notDetermined else {
                return settings.authorizationStatus == .authorized
            }

            do {
                let granted = try await UNUserNotificationCenter.current()
                    .requestAuthorization(options: [.alert, .sound, .badge])
                authorizationStatus = granted ? .authorized : .denied
                if granted {
                    UIApplication.shared.registerForRemoteNotifications()
                }
                return granted
            } catch {
                return false
            }
        }

        func markPrePromptShown() {
            prePromptShown = true
            hasCompletedPrePrompt = true
        }

        // Check if we should show pre-prompt (first login, not yet shown)
        var shouldShowPrePrompt: Bool {
            !prePromptShown && authorizationStatus == .notDetermined
        }

        // Opens iOS Settings for this app (for denied state)
        func openAppSettings() {
            if let url = URL(string: UIApplication.openSettingsURLString) {
                UIApplication.shared.open(url)
            }
        }

        // MARK: - UNUserNotificationCenterDelegate

        // D-10: Always show banner in foreground
        nonisolated func userNotificationCenter(
            _ center: UNUserNotificationCenter,
            willPresent notification: UNNotification
        ) async -> UNNotificationPresentationOptions {
            return [.banner, .sound, .badge]
        }

        // D-11: Deep-link on tap — handled by Plan 02
        nonisolated func userNotificationCenter(
            _ center: UNUserNotificationCenter,
            didReceive response: UNNotificationResponse
        ) async {
            let userInfo = response.notification.request.content.userInfo
            if let urlString = userInfo["instagram_url"] as? String,
               let url = URL(string: urlString) {
                await MainActor.run {
                    NotificationCenter.default.post(
                        name: .zenNotificationTapped,
                        object: nil,
                        userInfo: ["url": url]
                    )
                }
            }
        }
    }

    extension Notification.Name {
        static let zenNotificationTapped = Notification.Name("zenNotificationTapped")
    }
    ```

    Key design notes:
    - Per D-07: permission is NOT requested at launch. `refreshAuthorizationStatus()` only reads state.
    - Per D-10: `willPresent` returns `.banner` so foreground notifications always show.
    - Per D-11: `didReceive` posts a `Notification.Name` that ContentView will observe (Plan 02 wires deep-link navigation).
    - `nonisolated` on delegate methods because they're called by the system off-MainActor; the `async` variants handle the concurrency bridge.

        @State private var networkMonitor = NetworkMonitor()

            _ application: UIApplication,
            didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]? = nil

            return true

        func application(
            _ application: UIApplication,
            didRegisterForRemoteNotificationsWithDeviceToken deviceToken: Data

        func application(
            _ application: UIApplication,
            didFailToRegisterForRemoteNotificationsWithError error: Error
        ) {
            print("[APNs] Registration failed: \(error.localizedDescription)")
        }
    }

    Per D-09, a brief native pre-prompt that appears before the iOS system permission dialog. Implemented as a SwiftUI `.alert` modifier (not a full sheet -- keep it lightweight and calm). Copy should be intentional and aligned with ZenSocial's brand:

    Title: "Stay in the loop"
    Message: "ZenSocial can show you Instagram notifications as native banners -- no distractions, just what matters."
    Buttons: "Enable Notifications" (primary, triggers `NotificationManager.shared.requestPermission()`) and "Not Now" (cancel, marks pre-prompt as shown but does not request permission).

    This is NOT a standalone view file -- it is a ViewModifier or extension that can be applied to any view. Create it as:

    ```swift
    import SwiftUI

    struct NotificationPrePromptModifier: ViewModifier {
        @Binding var isPresented: Bool
        var onEnable: () async -> Void
        var onDismiss: () -> Void

        func body(content: Content) -> some View {
            content
                .alert("Stay in the loop", isPresented: $isPresented) {
                    Button("Enable Notifications") {
                        Task { await onEnable() }
                    }
                    Button("Not Now", role: .cancel) {
                        onDismiss()
                    }
                } message: {
                    Text("ZenSocial can show you Instagram notifications as native banners \u{2014} no distractions, just what matters.")
                }
        }
    }

    extension View {
        func notificationPrePrompt(
            isPresented: Binding<Bool>,
            onEnable: @escaping () async -> Void,
            onDismiss: @escaping () -> Void
        ) -> some View {
            modifier(NotificationPrePromptModifier(
                isPresented: isPresented,
                onEnable: onEnable,
                onDismiss: onDismiss
            ))
        }
    }
    ```
